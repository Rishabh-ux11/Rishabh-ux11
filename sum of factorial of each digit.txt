#include <stdio.h>
int main(){
    int num;
    int temp;
    int sum = 0;
    int digit;
    
    printf("enter a number:");
    scanf("%d", &num);
    
    temp=num;
    while(temp>0){
        int fact = 1;
        digit = temp %10;
        
        for(int i=1; i<=digit; i++){
            fact*=i;
            sum+=fact;
            temp = temp/10;
        }
        printf("Sum of factorial %d \n",sum);
    }
    return 0;
}
