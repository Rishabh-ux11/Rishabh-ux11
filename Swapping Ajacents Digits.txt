#include <stdio.h>
#include <string.h>
int main(){
    char num [100];
    
    printf("enter a number:");
    scanf("%s",num);
    
    int len= strlen(num);
    
    for (int i=0; i<len-1; i+=2){
        char temp = num[i];
        num[i] = num[i+1];
        num[i+1] = temp;
    }
    
    printf("After swapping  adjacent digits: %s\n",num);
    
    return 0;
}
