#include<stdio.h>
WITH *
int main(){
    int num;
    printf("enter a number to print the multipication table:");
    scanf("%d", &num);
    
    for(int i=1; i<15; i++){
        printf("%d * %d = %d\n", num,i,num*i);
    }
    return 0;
}


With out *
int main(){
    int num;
    printf("enter  a number to print a multiplication table:");
    scanf("%d",&num);
    
    for(int i=1; i<=10; i++){
        int result = 0;
        
        for (int j=0; j<i; j++){
            result +=num;
        }
        printf("%d x %d = %d\n", num, i, result);
    }
    
    return 0;
}
