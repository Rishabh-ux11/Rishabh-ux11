#include <stdio.h>

WHILE LOOP WITH FOR LOOP
int main(){
    int i = 1;
    int j;
  while(i<=3){
        for(j=1;j<=3;j++){
            printf("%d",j);
        }
         printf("\n");
       i++;
  }
    return 0;
}


FOR LOOP WITH WHILE LOOP
int main(){
    for (int i = 1; i<=3; i++){
        int j = 1;
         
        while(j<=3){
            printf("%d", j);
            j++;
        }
        printf("\n");
    }
    return 0;
}
