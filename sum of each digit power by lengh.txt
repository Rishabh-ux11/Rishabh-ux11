#include<stdio.h>
#include<math.h>
int main (){
    int num;
    int temp;
    int sum=0;
    int length= 0;
    
    printf("enter a number:");
    scanf("%d",&num);
    
    temp = num;
    while (temp>0){
        length++;
        temp = temp/10;
    }
    temp=num;
    while(temp>0){
        int digit = temp%10;
        sum+= pow(digit,length);
        temp = temp/10;
    }
    printf("sum of each digit raise to power%d = %d \n", length , sum);
    
    return 0;
}
