#include <stdio.h>
int main(){
    int base;
    int exponent;
    int result = 1;
    printf("enter base and exponent:");
    scanf("%d %d", &base, &exponent);
    
    for(int i = 0; i <= exponent; i++){
        result = result * base;
    }
    printf("%d^%d = %d \n", base, exponent, result);
}
