#include <stdio.h>
#include <math.h>
void main(){
    int num = 153;
    int count = 0,temp = num;
    while(temp > 0){
        count++;
        temp = num;
        int sum= 0;
        while(temp > 0){
            int digit = temp % 10;
            sum = sum += pow(digit, count);
            temp = temp/10;
        }
        if(num == sum){
            printf("%d is a Armstrong\n",num);
        }
        else{
            printf("%d is not a Armstrong\n",num);
        }
    }
}
